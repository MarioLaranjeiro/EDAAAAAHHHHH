#include <stdio.h>

void imprimetab(int tab[],int size)
{
    printf("\ntab[%d]={",size);
    for(int x=0;x<size;x++)
    {
        printf("%d,",tab[x]);
    }
    printf("}\n");
    
}

void invertetab(int tab[],int end,int start)
{
    int aux;
    if(start<end)
    {
        aux=tab[start];
        tab[start]=tab[end];
        tab[end]=aux;
        invertetab(tab,start+1,end-1);
    }
    else
    return;
}

int main() 
{  
    int n;
    int tab[n];
   
    printf("Insira o nr de elementos:");
    scanf("%d",&n);
    
    for(int i=0;i<n;i++)
    {
        printf("Insira o tab[%d]=",i);
        scanf("%d",&tab[i]);
    }
    imprimetab(tab,n);
    
    invertetab(tab,0,n-1);
    
    printf("\ntab invertida:");
    imprimetab(tab,n);
    return 0;
}
